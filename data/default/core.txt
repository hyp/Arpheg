id = 'core'

pipeline:
	id = 'debug.pipeline'
	glsl.vertexAttributes: 'position', 'colour'
	vertexShader:
		in vec3 position; in vec4 colour;
		uniform mat4 mvp;
		out vec4 vsColour;
		void main() {
			gl_Position = mvp*vec4(position,1); vsColour = colour;
		}
	pixelShader:
		in vec4 vsColour;
		out vec4 fragment;
		void main() { fragment = vsColour; }
